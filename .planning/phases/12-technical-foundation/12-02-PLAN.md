<task type="auto">
  <name>Task 2: Configure root layout with RTL and DirectionProvider</name>
  <files>
    website/src/app/layout.tsx
  </files>
  <action>
    Update website/src/app/layout.tsx with RTL configuration:

    ```tsx
    import type { Metadata } from "next";
    import { DirectionProvider } from "@radix-ui/react-direction";
    import "./globals.css";

    export const metadata: Metadata = {
      title: "Findo - Automatic Business Growth",
      description: "Autonomous lead capture, review management, and Google Business Profile optimization for Israeli SMBs",
    };

    export default function RootLayout({
      children,
    }: Readonly<{
      children: React.ReactNode;
    }>) {
      return (
        <html lang="he" dir="rtl">
          <body className="antialiased">
            <DirectionProvider dir="rtl">
              {children}
            </DirectionProvider>
          </body>
        </html>
      );
    }
    ```

    CRITICAL points:
    1. lang="he" for Hebrew language
    2. dir="rtl" on html element for browser RTL
    3. DirectionProvider INSIDE body (React context doesn't work on html)
    4. DirectionProvider dir="rtl" prop matches HTML attribute

    This setup ensures:
    - Native text direction is RTL
    - Scroll position starts from right
    - Future Radix/shadcn components will position correctly (dropdowns, tooltips)
  </action>
  <verify>
    Run `cd website && npm run build` passes.

    Inspect layout.tsx contains DirectionProvider wrapper inside body.
  </verify>
