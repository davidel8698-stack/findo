2. Create app/robots.ts:
   ```typescript
   import type { MetadataRoute } from "next";

   /**
    * Robots.txt for Findo sales website
    *
    * Rules:
    * - Allow all crawlers full access to public pages
    * - Block /api/ paths (none exist, but future-proofing)
    * - Block /admin/ paths (none exist, but future-proofing)
    * - Reference sitemap for discovery
    */
   export default function robots(): MetadataRoute.Robots {
     return {
       rules: [
         {
           userAgent: "*",
           allow: "/",
           disallow: ["/api/", "/admin/", "/_next/"],
         },
       ],
       sitemap: "https://findo.co.il/sitemap.xml",
     };
   }
   ```
